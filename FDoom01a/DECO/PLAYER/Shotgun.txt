ACTOR PerkShotgun : Shotgun replaces Shotgun
{	
	decal ShotChip
	States
	{
	Spawn:
		SHOT A 5
		loop
	Ready:
		SHTG A 1 A_WeaponReady(WRF_NOSECONDARY)
		loop
		SHTG A 1 A_WeaponReady
		loop
	Deselect:
		SHTG A 1 A_Lower
		wait
	Select:
		TNT1 AAAAAAAA 0 A_Raise
		SHTG HHCGGB 1 A_WeaponReady(WRF_NOFIRE)
		goto ready
	Fire: //fire 37 tics, refire 7 tics
		TNT1 A 0 A_ALertMonsters
		SHTG A 3
		TNT1 A 0 A_gunflash
		//SHTG A 3 A_FireBullets (5.6, 0, 7, 5, "EnShotPuff") //Exact equivalent of A_FireShotgun
		SHTG A 3 A_FireShotgun
		SHTG AEFBBGGC 1
		SHTG H 2
		SHTG H 3
		SHTG D 3
		SHTG HCGGBBFFE 1
		SHTG EAA 2
		SHTG A 7 A_ReFire
		Goto Ready
	Flash:
		SHTF B 3 Bright A_Light2
		SHTF A 2 Bright A_Light1
		Goto LightDone
	}
}
