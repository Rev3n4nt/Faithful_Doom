ACTOR BloxPlasmaRifle : PlasmaRifle replaces PlasmaRifle
{
   	States
   	{
	Spawn:
		PLRL A 5
		loop
	Ready:
		PLSG A 1 A_WeaponReady
		Loop
	Deselect:
		PLSG A 1 A_Lower
		Loop
	Select:
		PLSG A 1 A_Raise
		Loop
	Fire:
		PLSF B 1 Bright A_FirePlasma
		PLSF C 2 Bright
		TNT1 A 0 A_ReFire("fire2")
		PLSG a 1 Bright
		PLSG CCDDDE 1
		PLSG E 6
		PLSG EEBD 2
		PLSG CA 1
		Goto Ready
	Flash:
      	TNT1 A 1 A_Light1
      	Goto LightDone
		
		Bumper:
		TNT1 A 0
		stop
   	}
}
